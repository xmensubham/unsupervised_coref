\section{Introduction}

% ── Slide 1-1 ───────────────────────────────────────────────
\begin{frame}{What is Coreference Resolution?}
  \begin{bluebox}{Task Definition}
    Identify \textbf{all referring expressions} in a text that point to the
    \textbf{same real-world entity} and group them into
    \textit{coreference chains} (entity clusters).
  \end{bluebox}

  \bigskip
  \begin{exampleblock}{\normalsize Hindi Example}
    \centering\normalsize
    \textit{``\textbf{Ram} school gaya.\ \alert{Vah} bahut khush tha.\ \alert{Usne} doston se baat ki.''}\\[6pt]
    (Ram went to school.\ \alert{He} was happy.\ \alert{He} talked to his friends.)\\[8pt]
    \textbf{Entity Cluster:}\ \{Ram,\ Vah,\ Usne\}\ $\longrightarrow$\ all refer to the \emph{same person}
  \end{exampleblock}

  \bigskip
  \begin{block}{\normalsize Why It Matters — Downstream Applications}
    \centering\normalsize
    Machine Translation \quad$\cdot$\quad
    Question Answering \quad$\cdot$\quad
    Summarisation\\[4pt]
    Dialogue Systems \quad$\cdot$\quad
    Information Extraction \quad$\cdot$\quad
    Knowledge Graphs
  \end{block}
\end{frame}

% ── Slide 1-2 ───────────────────────────────────────────────
\begin{frame}{Why Hindi? The Challenge}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \begin{orangebox}{Hindi-Specific Challenges}
        \begin{itemize}\setlength\itemsep{5pt}
          \item \textbf{Morphological richness:}
                Nouns, verbs, pronouns inflect for gender, number, case
          \item \textbf{Pro-drop behaviour:}
                Subject pronouns are routinely omitted
          \item \textbf{Free word order:}
                SOV canonical, but deviations are common
          \item \textbf{Devanagari complexity:}
                Conjuncts, matras, zero-width joiners
          \item \textbf{Severe data scarcity:}
                No OntoNotes-scale resource exists for Hindi
        \end{itemize}
      \end{orangebox}

    \column{0.44\textwidth}
      \begin{greenbox}{The Scale of the Gap}
        \normalsize
        \begin{tabular}{@{}ll@{}}
          \toprule
          \textbf{Language} & \textbf{Coref Data}\\
          \midrule
          English & $\sim$10{,}000+ docs\\
          Hindi   & $\approx$ 0 usable\\
          \bottomrule
        \end{tabular}\\[10pt]
        \textbf{Hindi speakers:}\\
        530M native $+$ 100M L2\\[6pt]
        \alert{$\Rightarrow$ Critical NLP gap\\
        for 600M+ users}
      \end{greenbox}
  \end{columns}
\end{frame}

% ── Slide 1-3 ───────────────────────────────────────────────
\begin{frame}{Our Two-Phase Research Journey}

  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{orangebox}{Phase 1 — Scratch BERT (Failed)}
        \normalsize
        \textbf{Approach:} Train BERT from scratch on 172K Hindi\\
        Wikipedia docs; cluster mentions by cosine similarity\\[8pt]
        \textbf{Result:} \alert{Systematic failure}\\[4pt]
        Pronouns (\textit{vah, usne}) \emph{never} linked to\\
        correct antecedents at any threshold\\[8pt]
        \textbf{Root cause:} Coreference is \textbf{referential},\\
        not semantic — embedding similarity\\
        is structurally the wrong signal
      \end{orangebox}

    \column{0.48\textwidth}
      \begin{greenbox}{Phase 2 — Hybrid System (Succeeded)}
        \normalsize
        \textbf{Approach:} IndicBERTv2 contextual embeddings\\
        $+$ hand-crafted Hindi linguistic features\\
        $+$ Logistic Regression pairwise classifier\\
        $+$ Union-Find cluster reconstruction\\[8pt]
        \textbf{Dataset:} 76 manually annotated Hindi\\
        documents (548 mention spans)\\
        — created entirely from scratch\\[8pt]
        \textbf{Result:} $\approx$\textbf{82\%} pairwise accuracy\\
        \textcolor{green!55!black}{$+$11 pp from linguistic features}
      \end{greenbox}
  \end{columns}

  \bigskip\centering\small
  \textit{The Phase 1 failure is documented as a principled negative result —
  itself a research contribution.}
\end{frame}

        work on just 76 annotated documents?\\
        \hfill\textcolor{green!55!black}{Yes}\\[8pt]
        \textbf{Q3.} Do Hindi linguistic features add real\\
        value over embeddings alone?\\
        \hfill\textcolor{green!55!black}{+11 percentage points}

      \begin{bluebox}{Academic Value of Failure}

  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{\normalsize Primary Objectives}
        \begin{enumerate}\setlength\itemsep{7pt}
          \item Evaluate limits of unsupervised\\

          \item Build a \textbf{functional end-to-end}\\
                Hindi coreference pipeline
          \item Create a \textbf{manually annotated}\\
                Hindi coreference dataset\\
                (76 docs, 548 mention spans)
          \item Design a \textbf{hybrid architecture}\\
                combining IndicBERTv2 with\\
                hand-crafted Hindi features
        \end{enumerate}
      \end{block}

    \column{0.48\textwidth}

        \normalsize

        \begin{center}

          Dropout        & 0.1\\
          \bottomrule
        \end{tabular}
        \end{center}

      \end{block}
  \end{columns}
\end{frame}

% ── Slide 4-2 ───────────────────────────────────────────────
\begin{frame}{Phase 1 — Why It Failed}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{orangebox}{Failure Summary}
        \normalsize
        \renewcommand{\arraystretch}{1.40}
        \begin{tabular}{p{3.0cm} p{3.8cm}}
          \toprule
          \textbf{Failure Mode} & \textbf{Observation}\\
          \midrule
          Pronoun linking & Not linked at \textit{any} threshold\\
          Threshold tuning & No good precision-recall trade-off\\
          Spurious clusters & Low threshold merges wrong spans\\
          Entity name variation & Same entity unreliably merged\\
          Cross-sentence & No entity-tracking mechanism\\
          \bottomrule
        \end{tabular}
      \end{orangebox}

      \vspace{10pt}
      \begin{bluebox}{Key Metric}
        \centering\large
        Precision on pronoun-antecedent pairs\\
        at threshold 0.7:\\[6pt]
        \alert{\textbf{$<$ 30\%}}\\[4pt]
        \normalsize (barely above random chance)
      \end{bluebox}

    \column{0.46\textwidth}
      \begin{block}{\normalsize Root Cause Analysis}
        \normalsize
        \textbf{Ram} (proper noun) and \textbf{vah} (pronoun)\\
        have \alert{zero semantic similarity}\\
        — yet are coreferent.\\[10pt]
        Coreference is \textbf{referential}, not semantic.\\[10pt]
        Embedding cosine similarity detects\\
        semantic relatedness — which is\\
        \textbf{irrelevant} to referential identity.\\[10pt]
        This failure is \textit{systematic}: no amount\\
        of threshold tuning can fix it.
      \end{block}

      \vspace{8pt}
      \begin{greenbox}{Outcome}
        \normalsize
        Phase 1 formally discontinued.\\
        Findings documented as a\\
        \textbf{principled negative result}.
      \end{greenbox}
  \end{columns}
\end{frame}

% ── Slide 4-5 ───────────────────────────────────────────────
\begin{frame}{Hindi Linguistic Feature Engineering}
  \begin{columns}[T]
    \column{0.54\textwidth}
      \begin{block}{\normalsize Feature Vector for Mention Pair $(m_a,\,m_b)$}
        \normalsize
        \renewcommand{\arraystretch}{1.40}
        \begin{tabular}{p{3.2cm} p{4.2cm}}
          \toprule
          \textbf{Feature}          & \textbf{Description}\\
          \midrule
          Cosine similarity         & $\cos(\mathbf{m}_a,\mathbf{m}_b)$\\
          Embedding difference      & $|\mathbf{m}_a - \mathbf{m}_b|$ (reduced)\\
          Pronoun flag ($m_a$)      & 1 if \textit{vah, yah, main, hum, ve}\\
          Pronoun flag ($m_b$)      & 1 if $m_b$ is a Hindi pronoun\\
          Object pronoun            & 1 if \textit{use, inhe, unhe}\\
          Plural pronoun            & 1 if \textit{ve, hum, ye}\\
          Sentence distance         & Normalised gap between mentions\\
          Mention length ratio      & $|m_a|\,/\,|m_b|$ (tokens)\\
          Clause heuristic          & Same local subject clause?\\
          Head word match           & Identical stem?\\
          Mention compatibility     & Gender/number agreement signal\\
          \bottomrule
        \end{tabular}
      \end{block}

    \column{0.42\textwidth}
      \begin{bluebox}{Why Linguistic Features?}
        \normalsize
        Embedding similarity alone\\
        \textbf{cannot detect}:\\[6pt]
        \textit{Ram} (proper noun)\\
        $\leftrightarrow$ \textit{vah} (pronoun)\\[4pt]
        These share \alert{zero semantic overlap}\\
        — yet are clearly coreferent.\\[8pt]
        \textbf{Pronoun flag} signals:\\
        ``This mention is likely coreferent\\
        with a proximate nominal''\\[8pt]
        \textbf{Plural flag} prevents:\\
        \textit{ve} (they) from resolving\\
        to a singular antecedent
      \end{bluebox}

      \vspace{8pt}
      \begin{greenbox}{Ablation Preview}
        \centering\normalsize
        Embedding only\hfill$\approx$71\%\\
        + Linguistic features\hfill\textbf{$\approx$82\%}\\[4pt]
        \large\textcolor{green!55!black}{\textbf{+11 pp gain}}
      \end{greenbox}
  \end{columns}
\end{frame}

% ── Slide 4-6 ───────────────────────────────────────────────
\begin{frame}[fragile]{Cluster Reconstruction — Union-Find}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \begin{block}{\normalsize Python Implementation}
        \begin{lstlisting}
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))

    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[
                                self.parent[x]]
            x = self.parent[x]
        return x

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px != py:
            self.parent[px] = py

def build_clusters(mentions, pos_pairs):
    uf = UnionFind(len(mentions))
    for i, j in pos_pairs:
        uf.union(i, j)
    clusters = {}
    for idx in range(len(mentions)):
        clusters.setdefault(
            uf.find(idx), []).append(mentions[idx])
    return list(clusters.values())
        \end{lstlisting}
      \end{block}

    \column{0.44\textwidth}
      \begin{bluebox}{How It Works}
        \normalsize
        \textbf{Step 1:} Each mention starts\\
        in its own singleton set\\[6pt]
        \textbf{Step 2:} For each predicted\\
        positive pair $(m_a,m_b)$: merge sets\\[6pt]
        \textbf{Step 3:} Connected components\\
        become the entity clusters
      \end{bluebox}

      \vspace{8pt}
      \begin{orangebox}{Known Weakness}
        \normalsize
        \textbf{Transitive error propagation:}\\[4pt]
        If $m_a$--$m_b$ \emph{and} $m_b$--$m_c$ are both wrongly predicted coreferent, then $m_a$ and $m_c$ get merged even though they should not be.\\[4pt]
        One wrong prediction cascades.
      \end{orangebox}

        Language: Python 3.10\\
        Libraries: transformers, sklearn,\\
        \hspace{5.5em}sentencepiece, torch
      \end{block}
  \end{columns}
\end{frame}

          \item Documenting failure = valid research contribution
          \item Dataset creation = genuine community resource
        \end{itemize}

    \column{0.46\textwidth}
      \begin{orangebox}{Limitations (Honestly Reported)}
        \begin{enumerate}\setlength\itemsep{6pt}
          \item \textbf{Dataset size:}\\
                76 docs is small; generalisation\\
                beyond annotated domains is uncertain
          \item \textbf{No automatic mention detection:}\\
                System needs pre-supplied spans;\\
                not truly end-to-end
          \item \textbf{Classifier capacity:}\\
                Logistic regression has a linear ceiling
          \item \textbf{Transitive Union-Find errors:}\\
                One wrong pair cascades through\\
                the entire cluster
          \item \textbf{Evaluation coverage:}\\
                Only MUC + pairwise; no B$^3$/CEAF;\\
                full CoNLL F1 not computed
          \item \textbf{No inter-annotator agreement}\\
                (Cohen's Kappa not measured)
        \end{enumerate}
      \end{orangebox}
