\section{Introduction}

\begin{frame}{What is Coreference Resolution?}
  \begin{infobox}[Core Task Definition]
    Identify \textbf{all expressions} in a text that refer to the \textbf{same real-world entity}
    and group them into \textit{coreference chains} (entity clusters).
  \end{infobox}
  \vspace{0.4em}
  \begin{exampleblock}{Example in Hindi}
    \centering
    \textit{``Ram school gaya. \alert{Vah} bahut khush tha. \alert{Usne} apne doston se baat ki.''}\\[4pt]
    (Ram went to school. \alert{He} was very happy. \alert{He} talked to his friends.)\\[6pt]
    \textbf{Gold cluster:} \{Ram, Vah, Usne\}
  \end{exampleblock}
  \vspace{0.4em}
  \begin{block}{Why It Matters}
    \small
    \begin{columns}[T]
      \column{0.33\textwidth}
        \faRobot\ Machine Translation\\
        \faQuestionCircle\ Question Answering
      \column{0.33\textwidth}
        \faFileAlt\ Summarisation\\
        \faComments\ Dialogue Systems
      \column{0.33\textwidth}
        \faDatabase\ Information Extraction\\
        \faProjectDiagram\ Knowledge Graphs
    \end{columns}
  \end{block}
\end{frame}

\begin{frame}{Why Hindi? Why Is It Hard?}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \begin{warnbox}[Hindi-Specific Challenges]
        \small
        \begin{itemize}
          \item \textbf{Morphological richness:} Nouns, verbs, pronouns inflect for gender, number, case
          \item \textbf{Pro-drop:} Subject pronouns frequently omitted
          \item \textbf{Free word order:} SOV canonical but widely violated
          \item \textbf{Devanagari complexity:} Conjuncts, matras, ZWJ make tokenisation non-trivial
          \item \textbf{No large annotated corpus:} No OntoNotes equivalent for Hindi
        \end{itemize}
      \end{warnbox}
    \column{0.42\textwidth}
      \begin{successbox}[Scale of the Problem]
        \small
        \textbf{Hindi speakers:}\\
        530M native + 100M second-language\\[6pt]
        \textbf{English OntoNotes:}\\
        $\sim$10,000+ annotated documents\\[6pt]
        \textbf{Hindi coreference data:}\\
        Virtually none publicly available\\[6pt]
        \alert{$\Rightarrow$ Critical NLP gap for 600M+ users}
      \end{successbox}
  \end{columns}
\end{frame}

      \end{warnbox}
    \column{0.48\textwidth}
      \begin{successbox}[Research Questions We Answer]
        \small
        \begin{enumerate}
          \item Can embedding similarity alone resolve Hindi coreferences? \alert{(No --- and why)}
          \item Can a hybrid pipeline with IndicBERTv2 + hand-crafted Hindi features work on 76 documents? \textcolor{green!60!black}{(Yes)}
          \item Do linguistic features add value over embeddings alone? \textcolor{green!60!black}{(+11 pp accuracy)}
        \end{enumerate}
      \end{successbox}
      \vspace{0.3em}
      \begin{infobox}[Academic Contribution]
        \small
        Documenting the Phase~1 failure is itself a \textbf{research contribution} --- saves others from repeating a flawed approach
      \end{infobox}
  \end{columns}
\end{frame}

\begin{frame}{Objectives}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{Primary Objectives}
        \small
        \begin{enumerate}
          \item Empirically evaluate limits of unsupervised coreference via embedding clustering
          \item Build a \textbf{functional end-to-end} Hindi coreference pipeline
          \item Create a \textbf{manually annotated Hindi dataset} (76 docs, 548 mentions)
          \item Design a \textbf{hybrid architecture} combining IndicBERTv2 + linguistic features
        \end{enumerate}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Technical Objectives}
        \small
        \begin{itemize}

          \item Extract IndicBERTv2 mention embeddings (mean pooling)
          \item Engineer 11 Hindi-specific pairwise features
          \item Train Logistic Regression (5-fold CV, balanced weights)
          \item Union-Find cluster reconstruction
          \item Evaluate: Pairwise F1 + MUC coreference score
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Hindi Linguistic Feature Engineering}
  \begin{columns}[T]
    \column{0.54\textwidth}
      \begin{block}{Feature Vector for Mention Pair $(m_a, m_b)$}
        \scriptsize
        \begin{tabular}{p{2.8cm} p{4.0cm}}
          \toprule
          \textbf{Feature} & \textbf{Description}\\
          \midrule
          Cosine similarity & $\cos(\mathbf{m}_a, \mathbf{m}_b)$\\
          Embedding diff & $|\mathbf{m}_a - \mathbf{m}_b|$ (dim-reduced)\\
          Pronoun flag ($m_a$) & 1 if \textit{vah, yah, main, hum, tum, ve}\\
          Pronoun flag ($m_b$) & 1 if $m_b$ is a Hindi pronoun\\
          Object pronoun & 1 if \textit{use, inhe, unhe}\\
          Plural pronoun & 1 if \textit{ve, hum, ye}\\
          Sentence distance & Normalised sentence gap\\
          Mention length ratio & $|m_a| / |m_b|$ (tokens)\\
          Clause heuristic & Same local subject?\\
          Head word match & Identical / shared stem?\\
          Mention compatibility & Gender/number agreement heuristic\\
          \bottomrule
        \end{tabular}
      \end{block}
    \column{0.43\textwidth}
      \begin{infobox}[Why Linguistic Features?]
        \small
        Embedding similarity \textbf{cannot detect}:\\[4pt]
        \textbf{Ram} (proper noun) $\leftrightarrow$ \textbf{vah} (pronoun)\\
        = zero semantic overlap\\
        = coreferent anyway\\[6pt]
        Pronoun flag $\Rightarrow$ signals ``likely coreferent with proximate nominal''\\[6pt]
        Plural flag $\Rightarrow$ prevents \textit{ve} (they) resolving to singular antecedent
      \end{infobox}
      \vspace{0.4em}
      \begin{successbox}[Ablation Result Preview]
        \small
        Embedding only: $\approx$71\%\\
        + Linguistic features: $\approx$82\%\\[4pt]
        \alert{+11 percentage points} from feature engineering alone
      \end{successbox}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Union-Find Cluster Reconstruction}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \begin{block}{Algorithm}
        \lstset{language=Python, basicstyle=\ttfamily\tiny}
        \begin{lstlisting}
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))

    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]
            x = self.parent[x]
        return x

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px != py:
            self.parent[px] = py

def build_clusters(mentions, predicted_pairs):
    uf = UnionFind(len(mentions))
    for i, j in predicted_pairs:
        uf.union(i, j)
    clusters = {}
    for idx in range(len(mentions)):
        root = uf.find(idx)
        clusters.setdefault(root,[]).append(mentions[idx])
    return list(clusters.values())
        \end{lstlisting}
      \end{block}
    \column{0.44\textwidth}
      \begin{infobox}[How It Works]
        \small
        1. Each mention starts in its own singleton set\\[4pt]
        2. For each predicted positive pair $(m_a, m_b)$: merge sets\\[4pt]
        3. Final connected components = entity clusters
      \end{infobox}
      \vspace{0.4em}
      \begin{warnbox}[Known Weakness]
        \small
        \textbf{Transitive error propagation:}\\[4pt]
        If $m_a$--$m_b$ and $m_b$--$m_c$ are both incorrectly predicted as coreferent $\Rightarrow$ $m_a$ and $m_c$ are merged\\[4pt]
        One wrong prediction cascades through the cluster
      \end{warnbox}
      \vspace{0.3em}
      \begin{block}{Computational Environment}
        \scriptsize
        Platform: Google Colab / Kaggle\\
        GPU: T4 / P100 (free tier)\\
        Language: Python 3.10\\
        Libs: transformers, sklearn, sentencepiece, torch
      \end{block}
  \end{columns}
\end{frame}

\section{Experimental Results}

\begin{frame}{Phase 1 Results — Embedding Quality \& Failures}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{block}{Embedding Quality (Positive Signs)}
        \small
        $\bullet$ Semantically coherent nouns (body parts, country names, professions) showed \textbf{moderate clustering}\\[4pt]
        $\bullet$ Evidence MLM training captured distributional structure
      \end{block}
      \vspace{0.4em}
      \begin{warnbox}[Pronoun Behaviour (Critical Failure)]
        \small
        Hindi pronouns (\textit{vah, yah, main, hum}) formed a \textbf{loose cluster} --- but:\\[4pt]
        $\bullet$ No separation by number, case, or discourse role\\[4pt]
        $\bullet$ Precision on coreferent pronoun-antecedent pairs at threshold 0.7:\\[4pt]
        \centering\Large\alert{$< 30\%$}\\[4pt]
        \small Barely above chance for binary decision
      \end{warnbox}
    \column{0.47\textwidth}
      \begin{block}{Failure Summary Table}
        \scriptsize
        \begin{tabular}{p{2.3cm} p{3.1cm}}
          \toprule
          \textbf{Failure Mode} & \textbf{Observation}\\
          \midrule
          Pronoun-antecedent & Not linked at any threshold\\
          Threshold sensitivity & No good P-R trade-off\\
          Spurious clusters & Low threshold merges wrong spans\\
          Entity variation & Same entity, diff. surface form\\
          Cross-sentence & No entity-tracking mechanism\\
          \bottomrule
        \end{tabular}
      \end{block}
      \vspace{0.4em}
      \begin{infobox}[Conclusion of Phase 1]
        \small
        Failure is \textbf{systematic, not fixable by tuning}.\\[4pt]
        Reflects fundamental mismatch between task requirement and method capability.\\[4pt]
        \textbf{Phase 1 formally discontinued.}
      \end{infobox}
  \end{columns}
\end{frame}

\begin{frame}{Qualitative Examples}
  \begin{successbox}[Successful Resolution]
    \small
    \textbf{Input:} \textit{``Ram school gaya. Vah bahut khush tha. Usne apne doston se baat ki.''}\\
    (Ram went to school. He was very happy. He talked to his friends.)\\[4pt]
    \textbf{Gold cluster:} \{Ram, Vah, Usne\}
    \quad \textbf{Predicted:} \{Ram, Vah, Usne\} \textcolor{green!60!black}{\faCheckCircle}\\[4pt]
    \textit{Why it worked:} Pronoun flags for \textit{Vah} and \textit{Usne} fired; clause heuristic confirmed shared local subject with \textit{Ram}
  \end{successbox}

  \vspace{0.5em}

  \begin{warnbox}[Failure Case]
    \small
    \textbf{Input:} \textit{``Seema aur Priya dono school gayi. Vah bahut thak gayi thi.''}\\
    (Seema and Priya both went to school. She was very tired.)\\[4pt]
    \textbf{Gold cluster:} \{Seema aur Priya, Vah\} (collective antecedent)\\
    \textbf{Predicted:} \{Seema, Vah\} \textcolor{red!60!black}{\faTimesCircle} --- resolved to first-mentioned entity only\\[4pt]
    \textit{Why it failed:} Plurality check does not handle \textbf{conjoined noun phrases}; no mechanism for collective antecedent tracking
  \end{warnbox}
\end{frame}

\section{Conclusion \& Limitations}

\begin{frame}{Conclusion}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{block}{What We Achieved}
        \small
        $\bullet$ \textbf{Phase 1} (172K docs, unsupervised) --- failed systematically; documented the failure\\[4pt]
        $\bullet$ \textbf{Phase 2} (76 docs, supervised hybrid) --- functional coreference pipeline for Hindi\\[4pt]
        $\bullet$ \textbf{82\% overall pairwise accuracy}\\[4pt]
        $\bullet$ \textbf{+11 pp} from linguistic features\\[4pt]
        $\bullet$ MUC cluster-level improvement vs.\ Phase 1
      \end{block}
      \vspace{0.4em}
      \begin{infobox}[Lessons Learned]
        \small
        \textbf{L1:} Task formulation $>$ data volume\\[3pt]
        \textbf{L2:} Pretrained linguistic knowledge is invaluable even in tiny fine-tuning regimes\\[3pt]
        \textbf{L3:} Feature engineering still has real value in the transformer era at low data scales\\[3pt]
        \textbf{L4:} Documenting failure = valid research\\[3pt]
        \textbf{L5:} Dataset creation = genuine contribution
      \end{infobox}
    \column{0.47\textwidth}
      \begin{warnbox}[Limitations]
        \small
        \textbf{1. Dataset size:}\\
        76 docs is very small; generalisation uncertain beyond annotated domains\\[4pt]
        \textbf{2. No automatic mention detection:}\\
        System requires pre-supplied mention spans --- not truly end-to-end\\[4pt]
        \textbf{3. Classifier capacity:}\\
        Logistic regression = linear ceiling; non-linear interactions not captured\\[4pt]
        \textbf{4. Union-Find transitivity errors:}\\
        One wrong positive prediction cascades through clusters\\[4pt]
        \textbf{5. Evaluation coverage:}\\
        Only MUC + pairwise; no B$^3$ or CEAF; full CoNLL F1 not computed\\[4pt]
        \textbf{6. No inter-annotator agreement} measured (no Cohen's Kappa)
      \end{warnbox}
  \end{columns}
\end{frame}
